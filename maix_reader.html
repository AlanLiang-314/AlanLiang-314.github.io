閱讀器開發

Posted on Sat, Apr 2, 2022 python 
import lcd, image
lcd.init()
img = image.Image()
# image.font_load(image.UTF8, 16, 16, 0xA00000)
image.font_load(image.UTF8, 16, 16, '/sd/0xA00000_font_uincode_16_16_tblr.Dzk')
for i in range(12):
   img.draw_string(0, i*20, b'測試h123', x_spacing=2, mono_space=1) 
image.font_free()
lcd.display(img)
