[題解] 分組

Posted on Thu, Nov 4, 2021 題解 進階程設課程 
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int N,sum=0,maxn=-10e9,maxnum;
    string P;
    cin>>N>>P;
    int j=N,num=1;
    for(int i=P.size()-1;i>=0;i--){
        j--;
        sum+=P[i]-'0';
        if(j==0){
            if(sum>=maxn){
                maxn=sum;
                maxnum=num;
            }
            j=N;
            sum=0;
            num++;
        }
    }
		if(sum>=maxn){
       maxn=sum;
       maxnum=num;
    }
    cout<<maxnum<<" "<<maxn<<endl;
    return 0;
}
