Posted on Thu, Dec 23, 2021 進階程設課程 
#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s;
    map<string,string> m;
    queue<string> q;
    m["HELLO"]="ENGLISH";
    m["HOLA"]="SPANISH";
    m["HALLO"]="GERMAN";
    m["BONJOUR"]="FRENCH";
    m["CIAO"]="ITALIAN";
    m["ZDRAVSTVUJTE"]="RUSSIAN";
    while(cin>>s){
        if(s=="#") break;
        if(m.count(s)==1){
            q.push(m[s]);
        }else{
            q.push("UNKNOWN");
        }
    }
    int i=1;
    while(q.size()>0){
        cout<<"Case "<<i<<": "<<q.front()<<endl;
        q.pop();
        i++;
    }
    return 0;
}
