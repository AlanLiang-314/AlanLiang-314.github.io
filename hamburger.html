[題解] 美味蟹堡

Posted on Thu, Nov 4, 2021 題解 
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int N,K,st,ed,ans=0;
    cin>>N>>K;
    int S[N],W[N];
    for(int i=0;i<N;i++) cin>>S[i];
    for(int i=0;i<N;i++) cin>>W[i];
    st=0;
    for(int i=0;i<N;i++){
        while(W[st]==W[i]) i++;
        ed=i-1;
        sort(S+st,S+ed+1);
        //for(int j=0;j<N;j++) cout<<S[j]<<" ";
        //cout<<endl;
        //cout<<ed<<" "<<S[ed]<<endl;
        ans+=S[ed];
        st=i;
    }
    cout<<ans<<endl;

    return 0;
}
