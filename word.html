d518: 文字抄寫 II

Posted on Thu, Dec 23, 2021 進階程設課程 
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int num;
    string s;
    map<string,int> m;
    while(cin>>num){
        int index=1;
        for(int i=0;i<num;i++){
            cin>>s;
            if(m.count(s)!=0){
                cout<<"Old! "<<m[s]<<endl;
            }else{
                m[s]=index;
                cout<<"New! "<<index<<endl;
                index++;
            }
        }
        m.clear();
    }
    return 0;
}
