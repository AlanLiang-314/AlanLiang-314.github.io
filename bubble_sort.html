泡沫排序法

Posted on Thu, Oct 21, 2021 演算法 進階程設課程 
#include <iostream>

using namespace std;


int main()
{
    int a[9]={9,4,5,7,8,2,1,6,3};
    int len=sizeof(a)/sizeof(int);
    for(int i=0;i<len;i++){
        for(int j=0;j<len-i;j++){
            if(a[j+1]<a[j]) swap(a[j],a[j+1]);
        }
    }
    for(int i=0;i<len;i++) cout<<a[i]<<endl;
}
