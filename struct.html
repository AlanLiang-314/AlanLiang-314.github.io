Posted on Tue, Aug 24, 2021 C++

struct

struct類似python的tuple，可以將一些相關的變數綁在一起使用，方便閱讀。

#include <iostream>
#include <vector>

using namespace std;

struct classmate  //名稱
{
    int score;  //宣告變數
    int seatnumber;
};

int main(){
    classmate joy;
    joy.score = 100;
    joy.seatnumber = 53;
    cout<<"joy.score = "<<joy.score<<endl;
    cout<<"joy.seatnumber = "<<joy.seatnumber<<endl;

}
