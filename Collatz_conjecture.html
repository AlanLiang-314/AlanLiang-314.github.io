#include <iostream>
#include <vector>

using namespace std;

int dp[200000]={};

int find(long int a){
    int ans=1;
    while(a>1){
        if(a%2==0){
            a=a/2;
            ans++;
        }else{
            a=3*a+1;
            ans++;
        }
    }
    return ans;
}

int main(){
    int a;
    for(int i=0;i<10000000;i++){
        a=find(i);
        dp[a]++;
    }
    for(int i=0;i<150;i++){
        cout<<i<<": "<<dp[i]<<endl;
    }

}
