[題解] 考拉茲猜想路徑長

Posted on Sat, Oct 16, 2021 題解 C++

解法一：暴搜

直接計算一億以內正整數的路徑長，能保證路徑長23內的答案都正確

#include <iostream>
#include <vector>

using namespace std;

int dp[200000]={};

int find(long int a,int b){
    int ans=1;
    while(a>1&&ans<=b){
        if(a%2==0){
            a=a/2;
            ans++;
        }else{
            a=3*a+1;
            ans++;
        }
    }
    if(ans<=b){
        return ans;
    }else{
        return 0;
    }
}

int main(){
    int a;
    for(int i=1;i<100000000;i++){
        a=find(i,64);
        if(a) dp[a]++;
    }
    for(int i=0;i<=64;i++){
        cout<<i<<": "<<dp[i]<<endl;
    }

}
